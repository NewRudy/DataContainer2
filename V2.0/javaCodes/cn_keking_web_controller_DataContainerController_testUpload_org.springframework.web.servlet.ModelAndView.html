public class org.springframework.web.servlet.ModelAndView implements Serializable{

    private object view; //
    private unknow model; //
    private enum status; //
    private boolean cleared; //

    public object getView(){
        return view;
    }

    public void setView(object view){
        this.view = view;
    }

    public unknow getModel(){
        return model;
    }

    public void setModel(unknow model){
        this.model = model;
    }

    public enum getStatus(){
        return status;
    }

    public void setStatus(enum status){
        this.status = status;
    }

    public boolean getCleared(){
        return cleared;
    }

    public void setCleared(boolean cleared){
        this.cleared = cleared;
    }

}
